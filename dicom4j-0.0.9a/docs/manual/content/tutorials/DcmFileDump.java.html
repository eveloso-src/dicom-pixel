package samples.dicom4j.io.media;

import java.util.Iterator;

import org.apache.log4j.PropertyConfigurator;
import org.dicom4j.core.Dicom4j;
import org.dicom4j.data.elements.DataElement;
import org.dicom4j.io.media.DicomFile;
import org.slf4j.Logger;
import org.slf4j.LoggerFactory;

/**
 * sample class to show how to read a Dicom File
 *
 * @since 0.1
 * @author <a href="mailto:straahd@users.sourceforge.net">Laurent Lecomte 
 *
 */
public class DcmFileDump {
  
  /**
   * the Logger
   */
  private static final Logger flogger = LoggerFactory.getLogger(DcmFileDump.class);

  /**
   * 
   * @param args
   */
  public static void main(String[] args) {
    PropertyConfigurator.configure(".//samples-log4j.properties");
    flogger.info("DcmFileDump.start");
    try {
      Dicom4j.configure(); //we configure the framework
      DicomFile ldcm = new DicomFile( ".//mr-mono2-16bit.dcm" ); // we create the Dicom File
      ldcm.open(); // we open it (read the data)
      Iterator lIT = ldcm.getDataset().getIterator(); // we get an iterator to get elements
      int i=1;
      
      //We dump the File Meta Information
      flogger.info("\n" + ldcm.getFileMetaInformation() );
      
      // we parse all elements
      while (  lIT.hasNext() ) {
        //  we get the current Element (DataElement is the base class for all elements)
        DataElement lElement = (DataElement)lIT.next(); 
        flogger.info("Item " + i );
        // we Dump the tag's content
        flogger.info(" Tag ");
        flogger.info("  Group: " + lElement.getTag().getGroupAsString() );
        flogger.info("  Element: " + lElement.getTag().getElementAsString() );
        flogger.info("  Name: " + lElement.getTag().getName() );
        // We Dump the Element's content
        flogger.info(" Element");
        flogger.info("  VR: " + lElement.getValueRepresentation().getName() );
        flogger.info("  VM: " + lElement.getValueMultiplicity() );
        // we get the values as String[]
        String[] ldata = lElement.getStringValues();
        for (int j = 0; j < ldata.length; j++ ) {
          flogger.info("  Value " + new Integer(j+1) +" = " +  ldata[j]);
        }
        i++;
        flogger.info("");
      }
      ldcm.close(); //we close the File
      flogger.info("DcmFileDump.stop");
    } catch (Exception e) {
      flogger.error( e.getMessage() );
    }
  }

}
